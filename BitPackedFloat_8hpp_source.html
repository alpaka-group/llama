37  template<typename Integral>
38  LLAMA_FN_HOST_ACC_INLINE auto repackFloat(
39  Integral inFloat,
40  unsigned inMantissaBits,
41  unsigned inExponentBits,
42  unsigned outMantissaBits,
43  unsigned outExponentBits) -> Integral
44  {
45  const Integral inMantissaMask = (Integral{1} << inMantissaBits) - 1u;
46  const Integral inExponentMask = (Integral{1} << inExponentBits) - 1u;
47 
48  Integral inMantissa = inFloat & inMantissaMask;
49  const Integral inExponent = (inFloat >> inMantissaBits) & inExponentMask;
50  const Integral inSign = inFloat >> inExponentBits >> inMantissaBits;
51 
52  const Integral outExponentMask = (Integral{1} << outExponentBits) - 1u;
53  Integral outExponent;
54  if(inExponent == inExponentMask) [[LLAMA_UNLIKELY]]
55  outExponent = outExponentMask; // propagate +/- inf/nan
56  else if(inExponent == 0) [[LLAMA_UNLIKELY]]
57  outExponent = 0; // propagate -/+ zero
58  else
59  {
60  const int outExponentMax = 1 << (outExponentBits - 1); // NOLINT(hicpp-signed-bitwise)
61  const int outExponentMin = -outExponentMax + 1;
62  const int outExponentBias = outExponentMax - 1;
63  const int inExponentBias = (1 << (inExponentBits - 1)) - 1; // NOLINT(hicpp-signed-bitwise)
64 
65  const int exponent = static_cast<int>(inExponent) - inExponentBias;
66  const auto clampedExponent = std::clamp(exponent, outExponentMin, outExponentMax);
67  if(clampedExponent == outExponentMin || clampedExponent == outExponentMax)
68  inMantissa = 0; // when exponent changed, let value become inf and not nan
69  outExponent = clampedExponent + outExponentBias;
70  }
71  assert(outExponent < (1u << outExponentBits));
72 
73  const Integral packedMantissa = inMantissaBits > outMantissaBits
74  ? inMantissa >> (inMantissaBits - outMantissaBits)
75  : inMantissa << (outMantissaBits - inMantissaBits);
76  const Integral packedExponent = outExponent << outMantissaBits;
77  const Integral packedSign = inSign << outExponentBits << outMantissaBits;
78 
79  const auto outFloat = static_cast<Integral>(packedMantissa | packedExponent | packedSign);
80  return outFloat;
81  }
