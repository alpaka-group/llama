68  constexpr auto blobSize([[maybe_unused]] size_type blobIndex) const -> size_type
69  {
70  const auto flatSize = LinearizeArrayDimsFunctor{}.size(Base::extents());
71  if constexpr(blobs == Blobs::OnePerField)
72  {
73  constexpr auto typeSizes = []() constexpr
74  {
75  Array<size_type, blobCount> r{};
76  forEachLeafCoord<TRecordDim>([&r, i = 0](auto rc) mutable constexpr
77  { r[i++] = sizeof(GetType<TRecordDim, decltype(rc)>); });
78  return r;
79  }();
80  return flatSize * typeSizes[blobIndex];
81  }
82  else if constexpr(subArrayAlignment == SubArrayAlignment::Align)
83  {
84  size_type size = 0;
85  using FRD = typename Permuter::FlatRecordDim;
86  mp_for_each<mp_transform<mp_identity, FRD>>(
87  [&](auto ti) LLAMA_LAMBDA_INLINE
88  {
89  using FieldType = typename decltype(ti)::type;
90  size = roundUpToMultiple(size, static_cast<size_type>(alignof(FieldType)));
91  size += static_cast<size_type>(sizeof(FieldType)) * flatSize;
92  });
93  return size;
94  }
95  else
96  {
97  return flatSize * static_cast<size_type>(sizeOf<TRecordDim>);
98  }
99  }

102  static LLAMA_CONSTEVAL auto computeSubArrayOffsets()
103  {
104  using FRD = typename Permuter::FlatRecordDim;
105  constexpr auto staticFlatSize = LinearizeArrayDimsFunctor{}.size(TArrayExtents{});
106  constexpr auto subArrays = mp_size<FRD>::value;
107  Array<size_type, subArrays> r{};
108  // r[0] == 0, only compute the following offsets
109  mp_for_each<mp_iota_c<subArrays - 1>>(
110  [&](auto ic)
111  {
112  constexpr auto i = decltype(ic)::value;
113  r[i + 1] = r[i];
114  using ThisFieldType = mp_at_c<FRD, i>;
115  r[i + 1] += static_cast<size_type>(sizeof(ThisFieldType)) * staticFlatSize;
116  using NextFieldType = mp_at_c<FRD, i + 1>;
117  r[i + 1] = roundUpToMultiple(r[i + 1], static_cast<size_type>(alignof(NextFieldType)));
118  });
119  return r;
120  }

123  template<std::size_t... RecordCoords>
124  LLAMA_FN_HOST_ACC_INLINE constexpr auto blobNrAndOffset(
125  typename Base::ArrayIndex ai,
126  RecordCoord<RecordCoords...> = {}) const -> NrAndOffset<size_type>
127  {
128  const size_type elementOffset = LinearizeArrayDimsFunctor{}(ai, Base::extents())
129  * static_cast<size_type>(sizeof(GetType<TRecordDim, RecordCoord<RecordCoords...>>));
130  if constexpr(blobs == Blobs::OnePerField)
131  {
132  constexpr auto blob = flatRecordCoord<TRecordDim, RecordCoord<RecordCoords...>>;
133  return {blob, elementOffset};
134  }
135  else
136  {
137  constexpr std::size_t flatFieldIndex =
138 #if defined(__NVCC__) && __CUDACC_VER_MAJOR__ == 11 && __CUDACC_VER_MINOR__ <= 6
139  *& // mess with nvcc compiler state to workaround bug
140 #endif
141  Permuter::template permute<flatRecordCoord<TRecordDim, RecordCoord<RecordCoords...>>>;
142  const size_type flatSize = LinearizeArrayDimsFunctor{}.size(Base::extents());
143  using FRD = typename Permuter::FlatRecordDim;
144  if constexpr(subArrayAlignment == SubArrayAlignment::Align)
145  {
146  if constexpr(TArrayExtents::rankStatic == TArrayExtents::rank)
147  {
148  // full array extents are known statically, we can precompute the sub array offsets
149  constexpr auto subArrayOffsets = computeSubArrayOffsets();
150  return {0, subArrayOffsets[flatFieldIndex] + elementOffset};
151  }
152  else
153  {
154  // TODO(bgruber): we can take a shortcut here if we know that flatSize is a multiple of all
155  // type's alignment. We can also precompute a table of sub array starts (and maybe store it),
156  // or rely on the compiler it out of loops.
157  size_type offset = 0;
158  mp_for_each<mp_iota_c<flatFieldIndex>>(
159  [&](auto ic) LLAMA_LAMBDA_INLINE
160  {
161  constexpr auto i = decltype(ic)::value;
162  using ThisFieldType = mp_at_c<FRD, i>;
163  offset += static_cast<size_type>(sizeof(ThisFieldType)) * flatSize;
164  using NextFieldType = mp_at_c<FRD, i + 1>;
165  offset = roundUpToMultiple(offset, static_cast<size_type>(alignof(NextFieldType)));
166  });
167  offset += elementOffset;
168  return {0, offset};
169  }
170  }
171  else
172  {
173  const auto offset
174  = elementOffset + static_cast<size_type>(flatOffsetOf<FRD, flatFieldIndex, false>) * flatSize;
175  return {0, offset};
176  }
177  }
178  }
