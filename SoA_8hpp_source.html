43  constexpr auto blobSize([[maybe_unused]] size_type blobIndex) const -> size_type
44  {
45  const auto flatSize = LinearizeArrayDimsFunctor{}.size(Base::extents());
46  if constexpr(SeparateBuffers)
47  {
48  constexpr auto typeSizes = []() constexpr
49  {
50  Array<size_type, blobCount> r{};
51  forEachLeafCoord<TRecordDim>([&r, i = 0](auto rc) mutable constexpr {
52  r[i++] = sizeof(GetType<TRecordDim, decltype(rc)>);
53  });
54  return r;
55  }
56  ();
57  return flatSize * typeSizes[blobIndex];
58  }
59  else if constexpr(AlignSubArrays)
60  {
61  size_type size = 0;
62  using namespace boost::mp11;
63  using FRD = typename Flattener::FlatRecordDim;
64  mp_for_each<mp_transform<mp_identity, FRD>>(
65  [&](auto ti)

72  }

79  template<std::size_t... RecordCoords>
80  LLAMA_FN_HOST_ACC_INLINE constexpr auto blobNrAndOffset(
81  typename Base::ArrayIndex ad,
82  RecordCoord<RecordCoords...> = {}) const -> NrAndOffset<size_type>
83  {
84  if constexpr(SeparateBuffers)
85  {
86  constexpr auto blob = flatRecordCoord<TRecordDim, RecordCoord<RecordCoords...>>;

89  return {blob, offset};
90  }
91  else
92  {
93  const auto subArrayOffset = LinearizeArrayDimsFunctor{}(ad, Base::extents())
94  * static_cast<size_type>(sizeof(GetType<TRecordDim, RecordCoord<RecordCoords...>>));
95  constexpr std::size_t flatFieldIndex =
96 #ifdef __NVCC__
97  *& // mess with nvcc compiler state to workaround bug
98 #endif
99  Flattener::template flatIndex<RecordCoords...>;
100  const auto flatSize = LinearizeArrayDimsFunctor{}.size(Base::extents());
101  using FRD = typename Flattener::FlatRecordDim;
102  if constexpr(AlignSubArrays)
103  {
104  // TODO(bgruber): we can take a shortcut here if we know that flatSize is a multiple of all type's
105  // alignment. We can also precompute a table of sub array starts (and maybe store it), or rely on
106  // the compiler pulling it out of loops.
107  using namespace boost::mp11;
108  size_type offset = 0;
109  mp_for_each<mp_transform<mp_identity, mp_take_c<FRD, flatFieldIndex>>>(
110  [&](auto ti)
111  {
112  using FieldType = typename decltype(ti)::type;
113  offset = roundUpToMultiple(offset, static_cast<size_type>(alignof(FieldType)));
114  offset += static_cast<size_type>(sizeof(FieldType)) * flatSize;
115  });
116  offset = roundUpToMultiple(offset, static_cast<size_type>(alignof(mp_at_c<FRD, flatFieldIndex>)));
117  offset += subArrayOffset;
118  return {0, offset};
119  }
120  else
121  {
122  const auto offset
123  = subArrayOffset + static_cast<size_type>(flatOffsetOf<FRD, flatFieldIndex, false>) * flatSize;
124  return {0, offset};
125  }
126  }
127  }
