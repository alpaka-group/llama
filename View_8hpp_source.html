79  {
80  using View = View<Mapping, BlobType>;
81  using RecordDim = typename View::RecordDim;
82  forEachADCoord(
83  view.mapping().arrayDims(),
84  [&](typename View::ArrayDims ad)
85  {
86  if constexpr(isRecord<RecordDim> || internal::IsBoundedArray<RecordDim>::value)
87  forEachLeaf<RecordDim>(
88  [&](auto coord)
89  {
90  // TODO(bgruber): we could initialize computed fields if we can write to those. We could

94  });
95  else if constexpr(!isComputed<Mapping, RecordCoord<>>)
96  new(&view(ad)) RecordDim;
