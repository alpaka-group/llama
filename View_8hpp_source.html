90  template<typename Mapping, typename BlobType, typename Accessor, std::size_t... RCs>
91  LLAMA_FN_HOST_ACC_INLINE void constructField(
92  View<Mapping, BlobType, Accessor>& view,
93  typename Mapping::ArrayExtents::Index ai,
94  RecordCoord<RCs...> rc)
95  {
96  using FieldType = GetType<typename Mapping::RecordDim, decltype(rc)>;
97 
98  // this handles physical and computed mappings
99  if constexpr(sizeof...(RCs) == 0)
100  {
101  using RefType = decltype(view(ai));
102  if constexpr(isProxyReference<RefType>)
103  {
104  view(ai) = FieldType{};
105  }
106  else if constexpr(
107  std::is_lvalue_reference_v<RefType> && !std::is_const_v<std::remove_reference_t<RefType>>)
108  {
109  new(&view(ai)) FieldType{};
110  }
111  }
112  else
113  {
114  using RefType = decltype(view(ai)(rc));
115  if constexpr(isProxyReference<RefType>)
116  {
117  view(ai)(rc) = FieldType{};
118  }
119  else if constexpr(
120  std::is_lvalue_reference_v<RefType> && !std::is_const_v<std::remove_reference_t<RefType>>)
121  {
122  new(&view(ai)(rc)) FieldType{};
123  }
124  }
125  }

365  }
