114  template<typename Mapping, typename BlobType, typename Accessor>
115  LLAMA_FN_HOST_ACC_INLINE void constructFields(View<Mapping, BlobType, Accessor>& view)
116  {
117  using View = View<Mapping, BlobType, Accessor>;
118  using RecordDim = typename View::RecordDim;
119  forEachADCoord(
120  view.mapping().extents(),
121  [&]([[maybe_unused]] typename View::ArrayIndex ai)
122  {
123  if constexpr(isRecord<RecordDim> || internal::IsBoundedArray<RecordDim>::value)
124  {
125  forEachLeafCoord<RecordDim>(
126 #if defined(__NVCC__) && __CUDACC_VER_MAJOR__ == 11 && __CUDACC_VER_MINOR__ <= 4
127  internal::NvccWorkaroundLambda<View>{view, ai}
128 #else
129  [&](auto rc)
130  {
131  using FieldType = GetType<RecordDim, decltype(rc)>;
132  using RefType = decltype(view(ai)(rc));
133  // this handles physical and computed mappings
134  if constexpr(isProxyReference<RefType>)
135  {
136  view(ai)(rc) = FieldType{};
137  }
138  else if constexpr(
139  std::is_lvalue_reference_v<
140  RefType> && !std::is_const_v<std::remove_reference_t<RefType>>)
141  {
142  new(&view(ai)(rc)) FieldType;
143  }
144  }
145 #endif
146  );
147  }
148  else
149  {
150  // this handles physical and computed mappings
151  using RefType = decltype(view(ai));
152  if constexpr(isProxyReference<RefType>)
153  {
154  view(ai) = RecordDim{};
155  }
156  else if constexpr(
157  std::is_lvalue_reference_v<RefType> && !std::is_const_v<std::remove_reference_t<RefType>>)

159  new(&view(ai)) RecordDim;

372  template<typename Mapping, std::size_t... Coords, typename Blobs>
373  LLAMA_FN_HOST_ACC_INLINE auto mapToMemory(
374  Mapping& mapping,
375  typename Mapping::ArrayIndex ai,
376  RecordCoord<Coords...> rc,
377  Blobs& blobs) -> decltype(auto)
378  {
379  if constexpr(llama::isComputed<Mapping, RecordCoord<Coords...>>)
380  return mapping.compute(ai, rc, blobs);

384  using Type = GetType<typename Mapping::RecordDim, RecordCoord<Coords...>>;
385  LLAMA_BEGIN_SUPPRESS_HOST_DEVICE_WARNING
386  return reinterpret_cast<CopyConst<std::remove_reference_t<decltype(blobs[nr][offset])>, Type>&>(
387  blobs[nr][offset]);
388  LLAMA_END_SUPPRESS_HOST_DEVICE_WARNING
389  }
390  }
