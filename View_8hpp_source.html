78  {
79  using View = View<Mapping, BlobType>;
80  using RecordDim = typename View::RecordDim;
81  forEachADCoord(
82  view.mapping().arrayDims(),
83  [&](typename View::ArrayDims ad)
84  {
85  if constexpr(isRecord<RecordDim> || internal::IsBoundedArray<RecordDim>::value)
86  forEachLeaf<RecordDim>(
87  [&](auto coord)
88  {
89  // TODO(bgruber): we could initialize computed fields if we can write to those. We could
90  // test if the returned value can be cast to a T& and then attempt to write.

93  });
94  else if constexpr(!isComputed<Mapping, RecordCoord<>>)
95  new(&view(ad)) RecordDim;
