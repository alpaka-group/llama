126  {
127  using View = View<Mapping, BlobType, Accessor>;
128  using RecordDim = typename View::RecordDim;
129  forEachADCoord(
130  view.extents(),
131  [&]([[maybe_unused]] typename View::ArrayIndex ai)
132  {
133  if constexpr(isRecordDim<RecordDim>)
134  {
135  forEachLeafCoord<RecordDim>(
136 #if defined(__NVCC__) && __CUDACC_VER_MAJOR__ == 11 && __CUDACC_VER_MINOR__ <= 4
137  internal::NvccWorkaroundLambda<View>{view, ai}
138 #else
139  [&](auto rc)
140  {
141  using FieldType = GetType<RecordDim, decltype(rc)>;
142  using RefType = decltype(view(ai)(rc));
143  // this handles physical and computed mappings
144  if constexpr(isProxyReference<RefType>)
145  {
146  view(ai)(rc) = FieldType{};
147  }
148  else if constexpr(
149  std::is_lvalue_reference_v<RefType>
150  && !std::is_const_v<std::remove_reference_t<RefType>>)
151  {
152  new(&view(ai)(rc)) FieldType;
153  }
154  }
155 #endif
156  );
157  }
158  else
159  {
160  // this handles physical and computed mappings
161  using RefType = decltype(view(ai));
162  if constexpr(isProxyReference<RefType>)
163  {
164  view(ai) = RecordDim{};
165  }
166  else if constexpr(
167  std::is_lvalue_reference_v<RefType> && !std::is_const_v<std::remove_reference_t<RefType>>)

386  RecordCoord rc,

399  }
