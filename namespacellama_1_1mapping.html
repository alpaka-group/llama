inlineconstexpr
Initial value:
= []() constexpr
{
auto max = std::numeric_limits<std::size_t>::max();
forEachLeafCoord<RecordDim>(
[&](auto rc)
{
using AttributeType = GetType<RecordDim, decltype(rc)>;
max = std::min(max, VectorRegisterBits / (sizeof(AttributeType) * CHAR_BIT));
});
return max;
}
