112  template<std::size_t Dim>
113  LLAMA_FN_HOST_ACC_INLINE constexpr auto operator()(
114  const ArrayDims<Dim>& coord,
115  [[maybe_unused]] const ArrayDims<Dim>& size) const -> std::size_t
116  {
117  std::size_t r = 0;
118  for(std::size_t bit = 0; bit < (sizeof(std::size_t) * CHAR_BIT) / Dim; bit++)
119  for(std::size_t i = 0; i < Dim; i++)
120  r |= (coord[i] & (std::size_t{1} << bit)) << ((bit + 1) * (Dim - 1) - i);
