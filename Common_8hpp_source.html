102  struct LinearizeArrayDimsMorton
103  {
104  template<typename ArrayExtents>
105  LLAMA_FN_HOST_ACC_INLINE constexpr auto size(const ArrayExtents& extents) const ->
106  typename ArrayExtents::value_type
107  {
108  if constexpr(ArrayExtents::rank == 0)
109  return 0;
110  else
111  {
112  auto longest = extents[0];
113  for(int i = 1; i < static_cast<int>(ArrayExtents::rank); i++)
114  longest = std::max(longest, extents[i]);
115  const auto longestPO2 = bit_ceil(longest);
116  return intPow(longestPO2, static_cast<typename ArrayExtents::value_type>(ArrayExtents::rank));
117  }
118  }

123  template<typename ArrayExtents>
124  LLAMA_FN_HOST_ACC_INLINE constexpr auto operator()(
125  const typename ArrayExtents::Index& ai,
126  [[maybe_unused]] const ArrayExtents& extents) const -> typename ArrayExtents::value_type
127  {
128  using size_type = typename ArrayExtents::value_type;
129  constexpr auto rank = static_cast<size_type>(ArrayExtents::rank);
130  size_type r = 0;
131  for(size_type bit = 0; bit < (static_cast<size_type>(sizeof(size_type)) * CHAR_BIT) / rank; bit++)
132  for(size_type i = 0; i < rank; i++)
133  r |= (ai[i] & (size_type{1} << bit)) << ((bit + 1) * (rank - 1) - i);
134  return r;
135  }
136

139  LLAMA_FN_HOST_ACC_INLINE static constexpr auto bit_ceil(T n) -> T
140  {
141  T r = 1u;
142  while(r < n)
