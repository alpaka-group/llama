101  template<typename ArrayExtents>
102  LLAMA_FN_HOST_ACC_INLINE constexpr auto operator()(
103  const typename ArrayExtents::Index& ai,
104  [[maybe_unused]] const ArrayExtents& extents) const -> std::size_t
105  {
106  std::size_t r = 0;
107  for(std::size_t bit = 0; bit < (sizeof(std::size_t) * CHAR_BIT) / ArrayExtents::rank; bit++)
108  for(std::size_t i = 0; i < ArrayExtents::rank; i++)
109  r |= (ai[i] & (std::size_t{1} << bit)) << ((bit + 1) * (ArrayExtents::rank - 1) - i);
