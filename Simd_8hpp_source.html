79  {

206  template<typename T, typename Simd, typename SrcRC, typename DstRC>
207  LLAMA_FN_HOST_ACC_INLINE void loadSimdFromField(const T& srcRef, Simd& dstSimd, SrcRC srcRC, DstRC dstRC)
208  {
209  using FieldType = GetType<typename T::AccessibleRecordDim, SrcRC>;
210  using ElementSimd = std::decay_t<decltype(dstSimd(dstRC))>;
211  using Traits = SimdTraits<ElementSimd>;
212 
213  auto loadElementWise = [&]
214  {
215  auto b = ArrayIndexIterator{srcRef.view.extents(), srcRef.arrayIndex()};
216  for(std::size_t i = 0; i < Traits::lanes; i++)
217  reinterpret_cast<FieldType*>(&dstSimd(dstRC))[i]
218  = srcRef.view(*b++)(cat(typename T::BoundRecordCoord{}, srcRC));
219  };
220 
221  // TODO(bgruber): can we generalize the logic whether we can load a dstSimd from that mapping?
222  using Mapping = typename T::View::Mapping;
223  if constexpr(mapping::isSoA<Mapping>)
224  {
225  LLAMA_BEGIN_SUPPRESS_HOST_DEVICE_WARNING
226  dstSimd(dstRC) = Traits::loadUnaligned(&srcRef(srcRC));
227  LLAMA_END_SUPPRESS_HOST_DEVICE_WARNING
228  }
229  else if constexpr(mapping::isAoSoA<typename T::View::Mapping>)
230  {
231  // TODO(bgruber): this check is too strict
232  if(T::View::Mapping::ArrayExtents::rank == 1 && srcRef.arrayIndex()[0] % Traits::lanes == 0
233  && T::View::Mapping::lanes >= Traits::lanes)
234  {
235  LLAMA_BEGIN_SUPPRESS_HOST_DEVICE_WARNING
236  dstSimd(dstRC) = Traits::loadUnaligned(&srcRef(srcRC));
237  LLAMA_END_SUPPRESS_HOST_DEVICE_WARNING
238  }
239  else
240  loadElementWise();
241  }
242  else if constexpr(mapping::isAoS<Mapping>)
243  {
244  LLAMA_BEGIN_SUPPRESS_HOST_DEVICE_WARNING
245  dstSimd(dstRC) = Traits::gather(&srcRef(srcRC), aosStridedIndices<Mapping, FieldType, Traits::lanes>);
246  LLAMA_END_SUPPRESS_HOST_DEVICE_WARNING
247  }
248  else
249  loadElementWise();
250  }

306  LLAMA_EXPORT
307  template<typename T, typename Simd>
308  LLAMA_FN_HOST_ACC_INLINE void loadSimd(const T& srcRef, Simd& dstSimd)
309  {
310  // structured dstSimd type and record reference
311  if constexpr(isRecordRef<Simd> && isRecordRef<T>)
312  {
313  if constexpr(simdLanes<Simd> == simdLanes<T>) // fast path mainly for scalar SimdN<T, 1, ...>
314  dstSimd = srcRef;
315  else
316  {
317  using SrcARD = typename T::AccessibleRecordDim;
318  using DstArd = typename Simd::AccessibleRecordDim;
319  if constexpr(std::is_same_v<SrcARD, DstArd>)
320  {
321  forEachLeafCoord<SrcARD>([&](auto rc) LLAMA_LAMBDA_INLINE
322  { internal::loadSimdFromField(srcRef, dstSimd, rc, rc); });
323  }
324  else
325  {
326  forEachLeafCoord<SrcARD>(
327  [&](auto srcRC) LLAMA_LAMBDA_INLINE
328  {
329  using SrcInnerCoord = decltype(srcRC);
330  forEachLeafCoord<DstArd>(
331  [&](auto dstRC) LLAMA_LAMBDA_INLINE
332  {
333  using DstInnerCoord = decltype(dstRC);
334  if constexpr(hasSameTags<SrcARD, SrcInnerCoord, DstArd, DstInnerCoord>)
335  {
336  internal::loadSimdFromField(srcRef, dstSimd, srcRC, dstRC);
337  }
338  });
339  });
340  }
341  }
342  }
343  // unstructured dstSimd and reference type
344  else if constexpr(!isRecordRef<Simd> && !isRecordRef<T>)
345  {
346  LLAMA_BEGIN_SUPPRESS_HOST_DEVICE_WARNING
347  dstSimd = SimdTraits<Simd>::loadUnaligned(&srcRef);
348  LLAMA_END_SUPPRESS_HOST_DEVICE_WARNING
349  }
350  else
351  {
352  // TODO(bgruber): when could we get here? Is this always an error?
353  static_assert(sizeof(Simd) == 0, "Invalid combination of Simd type and reference type");
354  }
355  }

414  template<
415  std::size_t N,
416  template<typename, /* std::integral */ auto>
417  typename MakeSizedSimd,
418  typename View,
419  typename UnarySimdFunction>
420  void simdForEachN(View& view, UnarySimdFunction f)
421  {
422  using IndexType = typename View::Mapping::ArrayExtents::value_type;
423  const auto total = product(view.mapping().extents());
424  auto it = view.begin();
425  IndexType i = 0;
426  // simd loop
427  while(i + IndexType{N} <= total)
428  {
429  SimdN<typename View::RecordDim, N, MakeSizedSimd> simd;
430  loadSimd(*it, simd);
431  if constexpr(std::is_void_v<decltype(f(simd))>)
432  f(simd);
433  else
434  storeSimd(f(simd), *it);
435  i += IndexType{N};
436  it += IndexType{N};
437  }
438  // tail
439  while(i < total)
440  {
441  auto scalar = One<typename View::RecordDim>{*it};
442  if constexpr(std::is_void_v<decltype(f(scalar))>)
443  f(scalar);
444  else
445  *it = f(scalar);
446  ++i;
447  ++it;
448  }
449  }
