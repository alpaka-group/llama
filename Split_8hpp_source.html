13  template<typename... Fields, std::size_t FirstCoord, std::size_t... Coords>
14  auto partitionRecordDim(Record<Fields...>, RecordCoord<FirstCoord, Coords...>)
15  {
16  using Rec = Record<Fields...>;
17  if constexpr(sizeof...(Coords) == 0)
18  {
19  using Part1 = Record<mp_at_c<Rec, FirstCoord>>;
20  using Part2 = mp_erase_c<Rec, FirstCoord, FirstCoord + 1>;
21  return mp_list<Part1, Part2>{};
22  }
23  else
24  {
25  using FieldTag = GetTag<Rec, RecordCoord<FirstCoord>>;
26  using FieldType = GetType<Rec, RecordCoord<FirstCoord>>;
27  using InnerPartition = decltype(partitionRecordDim(FieldType{}, RecordCoord<Coords...>{}));
28  using Part1 = Record<Field<FieldTag, mp_first<InnerPartition>>>;
29  using Part2 = mp_replace_at_c<Rec, FirstCoord, Field<FieldTag, mp_second<InnerPartition>>>;
30  return mp_list<Part1, Part2>{};
31  }
32  }

179  template<std::size_t... RecordCoords>
180  LLAMA_FN_HOST_ACC_INLINE constexpr auto blobNrAndOffset(ArrayIndex ai, RecordCoord<RecordCoords...> = {}) const
181  -> NrAndOffset<size_type>
182  {
183  using Tags = GetTags<RecordDim, RecordCoord<RecordCoords...>>;
184 
185  if constexpr(internal::isSelected<RecordCoord<RecordCoords...>, NormalizedSelectorForMapping1>)
186  return mapping1.blobNrAndOffset(ai, GetCoordFromTags<RecordDim1, Tags>{});
187  else

189  auto nrAndOffset = mapping2.blobNrAndOffset(ai, GetCoordFromTags<RecordDim2, Tags>{});
190  if constexpr(SeparateBlobs)
191  nrAndOffset.nr += m1bc;
192  else
193  {
194  for(size_type i = 0; i < m1bc; i++)
195  nrAndOffset.offset += mapping1.blobSize(i);
196  }
197  return nrAndOffset;
198  }
