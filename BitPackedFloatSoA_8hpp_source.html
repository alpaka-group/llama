35  template<typename Integral>
36  auto repackFloat(
37  Integral inFloat,
38  unsigned inMantissaBits,
39  unsigned inExponentBits,
40  unsigned outMantissaBits,
41  unsigned outExponentBits) -> Integral
42  {
43  const Integral inMantissaMask = (Integral{1} << inMantissaBits) - 1u;
44  const Integral inExponentMask = (Integral{1} << inExponentBits) - 1u;
45 
46  Integral inMantissa = inFloat & inMantissaMask;
47  const Integral inExponent = (inFloat >> inMantissaBits) & inExponentMask;
48  const Integral inSign = inFloat >> inExponentBits >> inMantissaBits;
49 
50  const Integral outExponentMask = (Integral{1} << outExponentBits) - 1u;
51  Integral outExponent;
52  if(inExponent == inExponentMask) [[unlikely]]
53  outExponent = outExponentMask; // propagate +/- inf/nan
54  else if(inExponent == 0) [[unlikely]]
55  outExponent = 0; // propagate -/+ zero
56  else
57  {
58  const int outExponentMax = 1 << (outExponentBits - 1); // NOLINT(hicpp-signed-bitwise)
59  const int outExponentMin = -outExponentMax + 1;
60  const int outExponentBias = outExponentMax - 1;
61  const int inExponentBias = (1 << (inExponentBits - 1)) - 1; // NOLINT(hicpp-signed-bitwise)
62 
63  const int exponent = static_cast<int>(inExponent) - inExponentBias;
64  const auto clampedExponent = std::clamp(exponent, outExponentMin, outExponentMax);
65  if(clampedExponent == outExponentMin || clampedExponent == outExponentMax)
66  inMantissa = 0; // when exponent changed, let value become inf and not nan
67  outExponent = clampedExponent + outExponentBias;
68  }
69  assert(outExponent < (1u << outExponentBits));
70 
71  const Integral packedMantissa = inMantissaBits > outMantissaBits
72  ? inMantissa >> (inMantissaBits - outMantissaBits)
73  : inMantissa << (outMantissaBits - inMantissaBits);
74  const Integral packedExponent = outExponent << outMantissaBits;
75  const Integral packedSign = inSign << outExponentBits << outMantissaBits;
76 
77  const auto outFloat = static_cast<Integral>(packedMantissa | packedExponent | packedSign);
78  return outFloat;
79  }
