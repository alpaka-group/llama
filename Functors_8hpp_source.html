214  template<typename InternalTreeCoord, typename BasicCoord, typename Tree>
215  LLAMA_FN_HOST_ACC_INLINE auto basicCoordToResultCoordImpl(const BasicCoord& basicCoord, const Tree& tree) const
216  {
217  if constexpr(std::is_same_v<InternalTreeCoord, Tuple<>>)
218  {
219  if constexpr(std::is_same_v<BasicCoord, Tuple<>>)
220  return Tuple{};
221  else
222  {
223  const auto& childTree = get<BasicCoord::FirstElement::childIndex>(tree.childs);
224  const auto rt1 = basicCoord.first().arrayIndex / amount;
225  const auto rt2 = basicCoord.first().arrayIndex % amount * childTree.count
226  + basicCoord.rest().first().arrayIndex;
227  auto rt1Child = TreeCoordElement<BasicCoord::FirstElement::childIndex>{rt1};
228  auto rt2Child = TreeCoordElement<BasicCoord::RestTuple::FirstElement::childIndex>{rt2};
229  return tupleCat(Tuple{rt1Child}, tupleCat(Tuple{rt2Child}, popFront(basicCoord.rest())));
230  }
231  }
232  else
233  {
234  if constexpr(InternalTreeCoord::FirstElement::childIndex != BasicCoord::FirstElement::childIndex)
235  return basicCoord;
236  else
237  {
238  auto rest = basicCoordToResultCoordImpl<typename InternalTreeCoord::RestTuple>(
239  popFront(basicCoord),
240  get<BasicCoord::FirstElement::childIndex>(tree.childs));
241  return tupleCat(Tuple{basicCoord.first()}, rest);
242  }
243  }
244  }
