123  template<typename T = std::size_t, T... Sizes>
124  struct ArrayExtents : Array<T, ((Sizes == dyn) + ... + 0)>
125  {
126  static constexpr std::size_t rank = sizeof...(Sizes);
127  static constexpr auto rank_dynamic = ((Sizes == dyn) + ... + 0);
128  static constexpr auto rank_static = rank - rank_dynamic;
129 
130  using Index = ArrayIndex<T, rank>;
131  using value_type = T;
132 
133  template<std::size_t I>
134  LLAMA_FN_HOST_ACC_INLINE constexpr auto get() const -> value_type
135  {
136  using namespace boost::mp11;
137  using TypeList = mp_list_c<T, Sizes...>;
138  constexpr auto extent = mp_at_c<TypeList, I>::value;
139  if constexpr(extent != dyn)
140  return extent;
141  else
142  return static_cast<const Array<value_type, rank_dynamic>&>(
143  *this)[+mp_count<mp_take_c<TypeList, I>, std::integral_constant<T, dyn>>::value];
144  }
145 
146  LLAMA_FN_HOST_ACC_INLINE constexpr auto operator[](T i) const -> value_type
147  {
148  return boost::mp11::mp_with_index<rank>(i, [&](auto ic) { return get<decltype(ic)::value>(); });
149  }
