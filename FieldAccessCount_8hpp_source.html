172  template<std::size_t... RecordCoords, typename Blobs>
173  LLAMA_FN_HOST_ACC_INLINE auto compute(
174  typename Mapping::ArrayExtents::Index ai,
175  RecordCoord<RecordCoords...> rc,
176  Blobs& blobs) const -> decltype(auto)
177  {
178  static_assert(
179  !std::is_const_v<Blobs>,
180  "Cannot access (even just reading) data through FieldAccessCount from const blobs/view, since we need "
181  "to write "
182  "the access counts");
183 
184  auto& hits = fieldHits(blobs)[+flatRecordCoord<RecordDim, RecordCoord<RecordCoords...>>];
185  decltype(auto) ref = mapToMemory(inner(), ai, rc, blobs); // T& or proxy reference (value)
186  if constexpr(MyCodeHandlesProxyReferences)
187  {
188  using Value = GetType<RecordDim, decltype(rc)>;
189  using Ref = decltype(ref);
190  return internal::FieldAccessCountReference<Value, Ref, CountType>{std::forward<Ref>(ref), &hits};
191  }
192  else
193  {
194  internal::atomicInc(hits.memLocsComputed);
195  return ref;
196  }
197  }
