113  template<typename View, typename RecordCoord>
114  void boxesFromComputedField(
115  View& view,
116  typename View::Mapping::ArrayIndex ai,
117  RecordCoord rc,
118  std::vector<FieldBox<typename View::Mapping::ArrayIndex>>& infos)
119  {
120  using Mapping = typename View::Mapping;
121  using RecordDim = typename Mapping::RecordDim;
122 
123  auto emitInfo = [&](auto nrAndOffset, std::size_t size) {
124  infos.push_back({ai, internal::toVec(rc), recordCoordTags<RecordDim>(rc), nrAndOffset, size});
125  };
126 
127  using Type = GetType<RecordDim, decltype(rc)>;
128  // computed values can come from anywhere, so we can only apply heuristics
129  auto& blobs = view.storageBlobs;
130  auto&& ref = view.mapping().compute(ai, rc, blobs);
131 
132  // try to find the mapped address in one of the blobs
133  if constexpr(std::is_reference_v<decltype(ref)>)
134  {
135  auto address = reinterpret_cast<std::intptr_t>(&ref);
136  for(std::size_t i = 0; i < blobs.size(); i++)
137  {
138  // TODO(bgruber): this is UB, because we are comparing pointers from unrelated
139  // allocations
140  const auto front = reinterpret_cast<std::intptr_t>(&blobs[i][0]);
141  const auto back = reinterpret_cast<std::intptr_t>(&blobs[i][view.mapping().blobSize(i) - 1]);
142  if(front <= address && address <= back)
143  {
144  emitInfo(NrAndOffset{i, static_cast<std::size_t>(address - front)}, sizeof(Type));
145  return; // a mapping can only map to one location in the blobs
146  }
147  }
148  }
149 
150  if constexpr(std::is_default_constructible_v<Type>)
151  {
152  const auto infosBefore = infos.size();
153 
154  // try to observe written bytes
155  const auto pattern = std::is_same_v<Type, bool> ? std::uint8_t{0xFF} : std::uint8_t{0xAA};
156  fillBlobsWithPattern(view, pattern);
157  ref = Type{}; // a broad range of types is default constructible and should write
158  // zero bytes
159  auto wasTouched = [&](auto b) { return static_cast<std::uint8_t>(b) != pattern; };
160  for(std::size_t i = 0; i < Mapping::blobCount; i++)
161  {
162  const auto blobSize = view.mapping().blobSize(i);
163  const auto* begin = &blobs[i][0];
164  const auto* end = begin + blobSize;
165 
166  auto* searchBegin = begin;
167  while(true)
168  {
169  const auto* touchedBegin = std::find_if(searchBegin, end, wasTouched);
170  if(touchedBegin == end)
171  break;
172  const auto& touchedEnd = std::find_if_not(touchedBegin + 1, end, wasTouched);
173  emitInfo(
174  NrAndOffset{i, static_cast<std::size_t>(touchedBegin - begin)},
175  touchedEnd - touchedBegin);
176  if(touchedEnd == end)
177  break;
178  searchBegin = touchedEnd + 1;
179  }
180  }
181 
182  if(infosBefore != infos.size())
183  return;
184  }
185 
186  // if we come here, we could not find out where the value is coming from
187  emitInfo(NrAndOffset{Mapping::blobCount, std::size_t{0}}, sizeof(Type));
188  }

190  template<typename Mapping>
191  auto boxesFromMapping(const Mapping& mapping) -> std::vector<FieldBox<typename Mapping::ArrayIndex>>
192  {
193  std::vector<FieldBox<typename Mapping::ArrayIndex>> infos;
194 
195  std::optional<decltype(allocView(mapping))> view;
196  if constexpr(hasAnyComputedField<Mapping>())
197  view = allocView(mapping);
198 
199  using RecordDim = typename Mapping::RecordDim;
200  for(auto ai : ArrayIndexRange{mapping.extents()})
201  forEachLeafCoord<RecordDim>(
202  [&](auto rc)
203  {
204  using Type = GetType<RecordDim, decltype(rc)>;
205  if constexpr(llama::isComputed<Mapping, decltype(rc)>)
206  boxesFromComputedField(view.value(), ai, rc, infos);
207  else
208  {
209  const auto [nr, off] = mapping.blobNrAndOffset(ai, rc);
210  infos.push_back(
211  {ai,
212  internal::toVec(rc),
213  recordCoordTags<RecordDim>(rc),
214  {static_cast<std::size_t>(nr), static_cast<std::size_t>(off)},
215  sizeof(Type)});
216  }
217  });
218 
219  return infos;
220  }

222  template<typename ArrayIndex>
223  auto breakBoxes(std::vector<FieldBox<ArrayIndex>> boxes, std::size_t wrapByteCount)
224  -> std::vector<FieldBox<ArrayIndex>>
225  {
226  for(std::size_t i = 0; i < boxes.size(); i++)
227  {
228  auto& fb = boxes[i];
229  if(fb.nrAndOffset.offset / wrapByteCount != (fb.nrAndOffset.offset + fb.size - 1) / wrapByteCount)
230  {
231  const auto remainingSpace = wrapByteCount - fb.nrAndOffset.offset % wrapByteCount;
232  auto newFb = fb;
233  newFb.nrAndOffset.offset = fb.nrAndOffset.offset + remainingSpace;
234  newFb.size = fb.size - remainingSpace;
235  fb.size = remainingSpace;
236  boxes.push_back(newFb);
237  }
238  }
239  return boxes;
240  }

284  blobBlockWidth,
285  height,

287  blobYOffset[i] + height / 2,
288  name);

291  writeBlobHeader(i, mapping.blobSize(i), "Blob: " + std::to_string(i));

293  svg = fmt::format(
294  R"(<?xml version="1.0" encoding="UTF-8" standalone="no"?>
295 <svg width="{}" height="{}" xmlns="http://www.w3.org/2000/svg">
296  <style>
297  .label {{ font: {}px sans-serif; }}
298  </style>
299 )",
300  blobBlockWidth + wrapByteCount * byteSizeInPixel,
301  blobYOffset.back() == 0 ? 987654321 : blobYOffset.back() - byteSizeInPixel,
302  byteSizeInPixel / 2)
303  + svg;
304 
305  std::size_t computedSizeSoFar = 0;
306  std::size_t lastBlobNr = std::numeric_limits<std::size_t>::max();
307  std::size_t usedBytesInBlobSoFar = 0;
308  for(const auto& info : infos)
309  {
310  if(lastBlobNr != info.nrAndOffset.nr)
311  {
312  usedBytesInBlobSoFar = 0;
313  lastBlobNr = info.nrAndOffset.nr;
314  }
315 
316  const auto blobY = blobYOffset[info.nrAndOffset.nr];
317  const auto offset = [&]
318  {
319  if(info.nrAndOffset.nr < Mapping::blobCount)
320  return info.nrAndOffset.offset;
321 
322  const auto offset = computedSizeSoFar;
323  computedSizeSoFar += info.size;
324  return offset;
325  }();
326  auto x = (offset % wrapByteCount) * byteSizeInPixel + blobBlockWidth;
327  auto y = (offset / wrapByteCount) * byteSizeInPixel + blobY;
328  const auto fill = internal::color(info.recordCoord);
329  const auto width = byteSizeInPixel * info.size;
330 
331  const auto nextOffset = [&]
332  {
333  if(&info == &infos.back())
334  return std::numeric_limits<std::size_t>::max();
335  const auto& nextInfo = (&info)[1];
336  if(info.nrAndOffset.nr < Mapping::blobCount && info.nrAndOffset.nr == nextInfo.nrAndOffset.nr)
337  return nextInfo.nrAndOffset.offset;
338 
339  return std::numeric_limits<std::size_t>::max();
340  }();
341  const auto isOverlapped = offset < usedBytesInBlobSoFar || nextOffset < offset + info.size;
342  usedBytesInBlobSoFar = offset + info.size;
343 
344  constexpr auto cropBoxes = true;
345  if(cropBoxes)
346  {
347  svg += fmt::format(
348  R"(<svg x="{}" y="{}" width="{}" height="{}">
349 )",
350  x,
351  y,
352  width,
353  byteSizeInPixel);
354  x = 0;
355  y = 0;
356  }
357  svg += fmt::format(
358  R"(<rect x="{}" y="{}" width="{}" height="{}" fill="#{:X}" stroke="#000" fill-opacity="{}"/>
359 )",
360  x,
361  y,
362  width,
363  byteSizeInPixel,
364  fill,
365  isOverlapped ? 0.3 : 1.0);
366  for(std::size_t i = 1; i < info.size; i++)
367  {
368  svg += fmt::format(
369  R"(<line x1="{}" y1="{}" x2="{}" y2="{}" stroke="#777"/>
370 )",
371  x + i * byteSizeInPixel,
372  y + byteSizeInPixel * 2 / 3,
373  x + i * byteSizeInPixel,
374  y + byteSizeInPixel);
375  }
376  svg += fmt::format(
377  R"(<text x="{}" y="{}" fill="#000" text-anchor="middle" class="label">{} {}</text>
378 )",
379  x + width / 2,
380  y + byteSizeInPixel * 3 / 4,
381  internal::formatArrayIndex(info.arrayIndex),
382  internal::xmlEscape(std::string{info.recordTags}));
383  if(cropBoxes)
384  svg += R"(</svg>
385 )";
386  }
387 
388  if(hasAnyComputedField)
389  {
390  writeBlobHeader(Mapping::blobCount, computedSizeSoFar, "Comp.");
391 
392  // fix total SVG size
393  const auto i = svg.find("987654321");
394  assert(i != std::string::npos);
395  svg.replace(i, 9, std::to_string(blobYOffset.back() - byteSizeInPixel));
396  }
397 
398  svg += "</svg>";
399  return svg;
