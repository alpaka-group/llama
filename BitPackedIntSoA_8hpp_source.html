66  // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
67  LLAMA_FN_HOST_ACC_INLINE constexpr operator Integral() const
68  {
69  auto* p = ptr + bitOffset / bitsPerStoredIntegral;
70  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;
71  assert(p < endPtr);
72  auto v = p[0] >> innerBitOffset;
73 
74  const auto innerBitEndOffset = innerBitOffset + VHBits::value();
75  if(innerBitEndOffset <= bitsPerStoredIntegral)
76  {
77  const auto mask = (StoredIntegral{1} << VHBits::value()) - 1u;
78  v &= mask;
79  }
80  else
81  {
82  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
83  const auto bitsLoaded = bitsPerStoredIntegral - innerBitOffset;
84  const auto mask = (StoredIntegral{1} << excessBits) - 1u;
85  assert(p + 1 < endPtr);
86  v |= (p[1] & mask) << bitsLoaded;
87  }
88  if constexpr(std::is_signed_v<Integral>)
89  {
90  if(v & (StoredIntegral{1} << (VHBits::value() - 1)))
91  v |= ~StoredIntegral{0} << VHBits::value(); // sign extend
92  }
93  return static_cast<Integral>(v);
94  }

96  LLAMA_FN_HOST_ACC_INLINE constexpr auto operator=(Integral value) -> BitPackedIntRef&
97  {
98  // NOLINTNEXTLINE(bugprone-signed-char-misuse,cert-str34-c)
99  const auto unsignedValue = static_cast<StoredIntegral>(value);
100  const auto mask = (StoredIntegral{1} << VHBits::value()) - 1u;
101  StoredIntegral valueBits;
102  if constexpr(!std::is_signed_v<Integral>)
103  valueBits = unsignedValue & mask;
104  else
105  {
106  const auto magnitudeMask = (StoredIntegral{1} << (VHBits::value() - 1)) - 1u;
107  const auto isSigned = value < 0;
108  valueBits = (StoredIntegral{isSigned} << (VHBits::value() - 1)) | (unsignedValue & magnitudeMask);
109  }
110 
111  auto* p = ptr + bitOffset / bitsPerStoredIntegral;
112  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;
113  const auto clearMask = ~(mask << innerBitOffset);
114  assert(p < endPtr);
115  auto mem = p[0] & clearMask; // clear previous bits
116  mem |= valueBits << innerBitOffset; // write new bits
117  p[0] = mem;
118 
119  const auto innerBitEndOffset = innerBitOffset + VHBits::value();
120  if(innerBitEndOffset > bitsPerStoredIntegral)
121  {
122  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
123  const auto bitsWritten = bitsPerStoredIntegral - innerBitOffset;
124  const auto clearMask = ~((StoredIntegral{1} << excessBits) - 1u);
125  assert(p + 1 < endPtr);
126  auto mem = p[1] & clearMask; // clear previous bits
127  mem |= valueBits >> bitsWritten; // write new bits
128  p[1] = mem;
129  }
130 
131  return *this;
132  }
