72  LLAMA_FN_HOST_ACC_INLINE constexpr operator Integral() const
73  {
74  auto* p = ptr + bitOffset / bitsPerStoredIntegral;
75  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;
76  assert(p < endPtr);
77  auto v = p[0] >> innerBitOffset;
78 
79  const auto innerBitEndOffset = innerBitOffset + VHBits::value();
80  if(innerBitEndOffset <= bitsPerStoredIntegral)
81  {
82  const auto mask = (StoredIntegral{1} << VHBits::value()) - 1u;
83  v &= mask;
84  }
85  else
86  {
87  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
88  const auto bitsLoaded = bitsPerStoredIntegral - innerBitOffset;
89  const auto mask = (StoredIntegral{1} << excessBits) - 1u;
90  assert(p + 1 < endPtr);
91  v |= (p[1] & mask) << bitsLoaded;
92  }
93  if constexpr(std::is_signed_v<Integral>)
94  if(v & (StoredIntegral{1} << (VHBits::value() - 1)))
95  v |= ~StoredIntegral{0} << VHBits::value(); // sign extend
96  return static_cast<Integral>(v);

99  LLAMA_FN_HOST_ACC_INLINE constexpr auto operator=(Integral value) -> BitPackedIntRef&
100  {
101  const auto unsignedValue = static_cast<StoredIntegral>(value);
102  const auto mask = (StoredIntegral{1} << VHBits::value()) - 1u;
103  StoredIntegral valueBits;
104  if constexpr(!std::is_signed_v<Integral>)
105  valueBits = unsignedValue & mask;
106  else
107  {
108  const auto magnitudeMask = (StoredIntegral{1} << (VHBits::value() - 1)) - 1u;
109  const auto isSigned = value < 0;
110  valueBits = (StoredIntegral{isSigned} << (VHBits::value() - 1)) | (unsignedValue & magnitudeMask);
111  }
112 
113  auto* p = ptr + bitOffset / bitsPerStoredIntegral;
114  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;
115  const auto clearMask = ~(mask << innerBitOffset);
116  assert(p < endPtr);
117  auto mem = p[0] & clearMask; // clear previous bits
118  mem |= valueBits << innerBitOffset; // write new bits
119  p[0] = mem;
120 
121  const auto innerBitEndOffset = innerBitOffset + VHBits::value();
122  if(innerBitEndOffset > bitsPerStoredIntegral)
123  {
124  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
125  const auto bitsWritten = bitsPerStoredIntegral - innerBitOffset;
126  const auto clearMask = ~((StoredIntegral{1} << excessBits) - 1u);
127  assert(p + 1 < endPtr);
128  auto mem = p[1] & clearMask; // clear previous bits
129  mem |= valueBits >> bitsWritten; // write new bits
130  p[1] = mem;
131  }
132 
133  return *this;
134  }
