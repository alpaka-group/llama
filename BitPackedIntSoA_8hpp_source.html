68  LLAMA_FN_HOST_ACC_INLINE constexpr operator Integral() const
69  {
70  auto* p = ptr + bitOffset / bitsPerStoredIntegral;
71  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;

73  auto v = p[0] >> innerBitOffset;
74 
75  const auto innerBitEndOffset = innerBitOffset + bits;
76  if(innerBitEndOffset <= bitsPerStoredIntegral)
77  {
78  const auto mask = (StoredIntegral{1} << bits) - 1u;
79  v &= mask;
80  }
81  else
82  {
83  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
84  const auto bitsLoaded = bitsPerStoredIntegral - innerBitOffset;
85  const auto mask = (StoredIntegral{1} << excessBits) - 1u;
86  assert(p + 1 < endPtr);
87  v |= (p[1] & mask) << bitsLoaded;
88  }
89  if constexpr(std::is_signed_v<Integral>)
90  if(v & (StoredIntegral{1} << (bits - 1)))

95  LLAMA_FN_HOST_ACC_INLINE constexpr auto operator=(Integral value) -> BitPackedIntRef&
96  {
97  const auto unsignedValue = static_cast<StoredIntegral>(value);
98  const auto mask = (StoredIntegral{1} << bits) - 1u;
99  StoredIntegral valueBits;
100  if constexpr(!std::is_signed_v<Integral>)
101  valueBits = unsignedValue & mask;
102  else
103  {
104  const auto magnitudeMask = (StoredIntegral{1} << (bits - 1)) - 1u;
105  const auto isSigned = value < 0;
106  valueBits = (StoredIntegral{isSigned} << (bits - 1)) | (unsignedValue & magnitudeMask);
107  }
108 
109  auto* p = ptr + bitOffset / bitsPerStoredIntegral;
110  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;
111  const auto clearMask = ~(mask << innerBitOffset);
112  assert(p < endPtr);
113  auto mem = p[0] & clearMask; // clear previous bits
114  mem |= valueBits << innerBitOffset; // write new bits
115  p[0] = mem;
116 
117  const auto innerBitEndOffset = innerBitOffset + bits;
118  if(innerBitEndOffset > bitsPerStoredIntegral)
119  {
120  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
121  const auto bitsWritten = bitsPerStoredIntegral - innerBitOffset;
122  const auto clearMask = ~((StoredIntegral{1} << excessBits) - 1u);
123  assert(p + 1 < endPtr);
124  auto mem = p[1] & clearMask; // clear previous bits
125  mem |= valueBits >> bitsWritten; // write new bits
126  p[1] = mem;
127  }
128 
129  return *this;
130  }
