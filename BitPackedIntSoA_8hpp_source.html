22  template<typename Integral, typename StoredIntegral>
23  LLAMA_FN_HOST_ACC_INLINE constexpr auto bitunpack(
24  const StoredIntegral* ptr,
25  StoredIntegral bitOffset,
26  StoredIntegral bitCount) -> Integral
27  {
28  constexpr auto bitsPerIntegral = static_cast<StoredIntegral>(sizeof(Integral) * CHAR_BIT);
29  constexpr auto bitsPerStoredIntegral = static_cast<StoredIntegral>(sizeof(StoredIntegral) * CHAR_BIT);
30  static_assert(bitsPerIntegral <= bitsPerStoredIntegral);
31  assert(bitCount > 0 && bitCount <= bitsPerStoredIntegral);
32 #ifdef __clang__
33  // this is necessary to silence the clang static analyzer
34  __builtin_assume(bitCount > 0 && bitCount <= bitsPerStoredIntegral);
35 #endif
36 
37  const auto* p = ptr + bitOffset / bitsPerStoredIntegral;
38  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;
39  // assert(p < endPtr);
40  auto v = p[0] >> innerBitOffset;
41 
42  const auto innerBitEndOffset = innerBitOffset + bitCount;
43  if(innerBitEndOffset <= bitsPerStoredIntegral)
44  {
45  const auto mask = makeMask(bitCount);
46  v &= mask;
47  }
48  else
49  {
50  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
51  const auto bitsLoaded = bitsPerStoredIntegral - innerBitOffset;
52  const auto mask = makeMask(excessBits);
53  // assert(p + 1 < endPtr);
54  v |= (p[1] & mask) << bitsLoaded;
55  }
56  if constexpr(std::is_signed_v<Integral>)
57  {
58  // perform sign extension
59  if((v & (StoredIntegral{1} << (bitCount - 1))) && bitCount < bitsPerStoredIntegral)
60  v |= ~StoredIntegral{0} << bitCount;
61  }
62  return static_cast<Integral>(v);
63  }

65  template<typename Integral, typename StoredIntegral>
66  LLAMA_FN_HOST_ACC_INLINE constexpr void bitpack(
67  StoredIntegral* ptr,
68  StoredIntegral bitOffset,
69  StoredIntegral bitCount,
70  Integral value)
71  {
72  constexpr auto bitsPerIntegral = static_cast<StoredIntegral>(sizeof(Integral) * CHAR_BIT);
73  constexpr auto bitsPerStoredIntegral = static_cast<StoredIntegral>(sizeof(StoredIntegral) * CHAR_BIT);
74  static_assert(bitsPerIntegral <= bitsPerStoredIntegral);
75  assert(bitCount > 0 && bitCount <= bitsPerStoredIntegral);
76 #ifdef __clang__
77  // this is necessary to silence the clang static analyzer
78  __builtin_assume(bitCount > 0 && bitCount <= bitsPerStoredIntegral);
79 #endif
80 
81  // NOLINTNEXTLINE(bugprone-signed-char-misuse,cert-str34-c)
82  const auto unsignedValue = static_cast<StoredIntegral>(value);
83  const auto mask = makeMask(bitCount);
84  StoredIntegral valueBits;
85  if constexpr(!std::is_signed_v<Integral>)
86  valueBits = unsignedValue & mask;
87  else
88  {
89  const auto magnitudeMask = makeMask(bitCount - 1);
90  const auto isSigned = value < 0;
91  valueBits = (StoredIntegral{isSigned} << (bitCount - 1)) | (unsignedValue & magnitudeMask);
92  }
93 
94  auto* p = ptr + bitOffset / bitsPerStoredIntegral;
95  const auto innerBitOffset = bitOffset % bitsPerStoredIntegral;
96 
97  {
98  const auto clearMask = ~(mask << innerBitOffset);
99  // assert(p < endPtr);
100  auto mem = p[0] & clearMask; // clear previous bits
101  mem |= valueBits << innerBitOffset; // write new bits
102  p[0] = mem;
103  }
104 
105  const auto innerBitEndOffset = innerBitOffset + bitCount;
106  if(innerBitEndOffset > bitsPerStoredIntegral)
107  {
108  const auto excessBits = innerBitEndOffset - bitsPerStoredIntegral;
109  const auto bitsWritten = bitsPerStoredIntegral - innerBitOffset;
110  const auto clearMask = ~makeMask(excessBits);
111  // assert(p + 1 < endPtr);
112  auto mem = p[1] & clearMask; // clear previous bits
113  mem |= valueBits >> bitsWritten; // write new bits
114  p[1] = mem;
115  }
116  }
