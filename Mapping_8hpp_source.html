147  template<typename Tree, typename... Coords>
148  LLAMA_FN_HOST_ACC_INLINE auto getTreeBlobByte(const Tree& tree, const Tuple<Coords...>& treeCoord)
149  -> std::size_t
150  {
151  const auto firstArrayIndex = treeCoord.first().arrayIndex;
152  if constexpr(sizeof...(Coords) > 1)
153  {
154  constexpr auto firstChildIndex = decltype(treeCoord.first().childIndex)::value;
155  return getTreeBlobSize(tree.childs, firstArrayIndex)
156  + sumChildrenSmallerThan<firstChildIndex>(
157  tree,
158  std::make_index_sequence<std::tuple_size_v<typename Tree::ChildrenTuple>>{})
159  + getTreeBlobByte(get<firstChildIndex>(tree.childs), treeCoord.rest());
160  }
161  else
162  return sizeof(typename Tree::Type) * firstArrayIndex;
163  }
