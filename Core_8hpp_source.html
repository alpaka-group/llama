588  {
589  std::size_t size = 0;
590  std::size_t maxAlign = 0; // NOLINT(misc-const-correctness)
591  mp_for_each<mp_transform<mp_identity, TypeList>>(
592  [&](auto e) constexpr
593  {
594  using T = typename decltype(e)::type;

597  size = roundUpToMultiple(size, alignof(T));
598  maxAlign = std::max(maxAlign, alignof(T));
599  }
600  // NOLINTNEXTLINE(readability-misleading-indentation)
601  size += sizeof(T);
602  });
603 
604  // final padding, so next struct can start right away
605  if constexpr(Align && IncludeTailPadding)
606  if(maxAlign > 0)
607  size = roundUpToMultiple(size, maxAlign); // TODO(bgruber): we could use flatAlignOf<TypeList>
608  // here, at the cost of more template instantiations
