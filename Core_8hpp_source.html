475  {
476  std::size_t size = 0;
477  std::size_t maxAlign = 0; // NOLINT(misc-const-correctness)
478  mp_for_each<mp_transform<mp_identity, TypeList>>(
479  [&](auto e) constexpr
480  {

485  maxAlign = std::max(maxAlign, alignof(T));
486  }
487  // NOLINTNEXTLINE(readability-misleading-indentation)
488  size += sizeof(T);
489  });
490 
491  // final padding, so next struct can start right away
492  if constexpr(Align && IncludeTailPadding)
493  size = roundUpToMultiple(size, maxAlign); // TODO(bgruber): we could use flatAlignOf<TypeList> here, at
494  // the cost of more template instantiations
495  return size;
496  }
