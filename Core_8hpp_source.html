672  template<
673  typename... FieldsA,
674  typename FieldB,
675  typename... FieldsB,
676  auto Pos = FindFieldByTag<Record<FieldsA...>, GetFieldTag<FieldB>>::value>
677  auto mergeRecordDimsImpl(
678  boost::mp11::mp_identity<Record<FieldsA...>>,
679  boost::mp11::mp_identity<Record<FieldB, FieldsB...>>)
680  {
681  using namespace boost::mp11;
682  if constexpr(Pos == sizeof...(FieldsA))
683  {
684  return mergeRecordDimsImpl(
685  mp_identity<Record<FieldsA..., FieldB>>{},
686  mp_identity<Record<FieldsB...>>{});
687  }
688  else
689  {

691  using NewFieldA = Field<
692  GetFieldTag<OldFieldA>,
693  typename decltype(mergeRecordDimsImpl(
694  mp_identity<GetFieldType<OldFieldA>>{},
695  mp_identity<GetFieldType<FieldB>>{}))::type>;
696  using NewRecordA = mp_replace_at_c<Record<FieldsA...>, Pos, NewFieldA>;
697  return mergeRecordDimsImpl(mp_identity<NewRecordA>{}, mp_identity<Record<FieldsB...>>{});
