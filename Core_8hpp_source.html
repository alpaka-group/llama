458  {
459  using namespace boost::mp11;

463  mp_for_each<mp_transform<mp_identity, TypeList>>([&](auto e) constexpr
464  {

469  maxAlign = std::max(maxAlign, alignof(T));

472  size += sizeof(T);
473  });
474 
475  // final padding, so next struct can start right away
476  if constexpr(Align && IncludeTailPadding)

478  // the cost of more template instantiations
