199  template<typename T>
200  LLAMA_FN_HOST_ACC_INLINE auto insert(iterator pos, T&& t) -> iterator
201  {
202  const auto i = pos - begin();
203  reserve(m_size + 1); // might invalidate pos
204  pos = begin() + i;
205  std::copy_backward(pos, end(), end() + 1);
206  m_view[i] = std::forward<T>(t);
207  m_size++;
208  return pos;
