38  template <
39  typename Functor,
40  typename LeftRecord,
41  typename RightView,
42  typename RightBoundRecordDim,
43  bool RightOwnView>
44  LLAMA_FN_HOST_ACC_INLINE auto virtualRecordArithOperator(
45  LeftRecord& left,
46  const VirtualRecord<RightView, RightBoundRecordDim, RightOwnView>& right) -> LeftRecord&
47  {
48  using RightRecord = VirtualRecord<RightView, RightBoundRecordDim, RightOwnView>;
49  // if the record dimension left and right is the same, a single loop is enough and no tag check is needed.
50  // this safes a lot of compilation time.
51  if constexpr (std::is_same_v<
52  typename LeftRecord::AccessibleRecordDim,
53  typename RightRecord::AccessibleRecordDim>)
54  {
55  forEachLeaf<typename LeftRecord::AccessibleRecordDim>([&](auto coord) LLAMA_LAMBDA_INLINE
56  { Functor{}(left(coord), right(coord)); });
57  }
58  else
59  {
60  forEachLeaf<typename LeftRecord::AccessibleRecordDim>(
61  [&](auto leftCoord) LLAMA_LAMBDA_INLINE
62  {
63  using LeftInnerCoord = decltype(leftCoord);
64  forEachLeaf<typename RightRecord::AccessibleRecordDim>(
65  [&](auto rightCoord) LLAMA_LAMBDA_INLINE
66  {
67  using RightInnerCoord = decltype(rightCoord);
68  if constexpr (hasSameTags<
69  typename LeftRecord::AccessibleRecordDim,
70  LeftInnerCoord,
71  typename RightRecord::AccessibleRecordDim,
72  RightInnerCoord>)
73  {
74  Functor{}(left(leftCoord), right(rightCoord));
75  }
76  });
77  });
78  }
79  return left;
80  }

135  template <typename Functor, typename LeftRecord, typename T>
136  LLAMA_FN_HOST_ACC_INLINE auto virtualRecordRelOperator(const LeftRecord& left, const T& right) -> bool
137  {
138  bool result = true;
139  forEachLeaf<typename LeftRecord::AccessibleRecordDim>(
140  [&](auto leftCoord) LLAMA_LAMBDA_INLINE {
141  result &= Functor{}(
142  left(leftCoord),
143  static_cast<std::remove_reference_t<decltype(left(leftCoord))>>(right));
144  });
145  return result;
