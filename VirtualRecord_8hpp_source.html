35  template<
36  typename Functor,
37  typename LeftRecord,
38  typename RightView,
39  typename RightBoundRecordDim,
40  bool RightOwnView>
41  LLAMA_FN_HOST_ACC_INLINE auto virtualRecordArithOperator(
42  LeftRecord& left,
43  const VirtualRecord<RightView, RightBoundRecordDim, RightOwnView>& right) -> LeftRecord&
44  {
45  using RightRecord = VirtualRecord<RightView, RightBoundRecordDim, RightOwnView>;
46  // if the record dimension left and right is the same, a single loop is enough and no tag check is needed.
47  // this safes a lot of compilation time.
48  if constexpr(std::is_same_v<
49  typename LeftRecord::AccessibleRecordDim,
50  typename RightRecord::AccessibleRecordDim>)
51  {
52  forEachLeafCoord<typename LeftRecord::AccessibleRecordDim>([&](auto rc) LLAMA_LAMBDA_INLINE
53  { Functor{}(left(rc), right(rc)); });
54  }
55  else
56  {
57  forEachLeafCoord<typename LeftRecord::AccessibleRecordDim>(
58  [&](auto leftRC) LLAMA_LAMBDA_INLINE
59  {
60  using LeftInnerCoord = decltype(leftRC);
61  forEachLeafCoord<typename RightRecord::AccessibleRecordDim>(
62  [&](auto rightRC) LLAMA_LAMBDA_INLINE
63  {
64  using RightInnerCoord = decltype(rightRC);
65  if constexpr(hasSameTags<
66  typename LeftRecord::AccessibleRecordDim,
67  LeftInnerCoord,
68  typename RightRecord::AccessibleRecordDim,
69  RightInnerCoord>)
70  {
71  Functor{}(left(leftRC), right(rightRC));
72  }
73  });
74  });
75  }
76  return left;
77  }

271  else
272  static_assert(
273  dependentFalse<T1, T2>,
274  "Elements to assign are not tuple/tuple or non-tuple/non-tuple.");
