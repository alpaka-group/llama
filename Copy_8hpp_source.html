138  template<typename SrcMapping, typename SrcBlob, typename DstMapping, typename DstBlob>
139  void aosoaCommonBlockCopy(
140  const View<SrcMapping, SrcBlob>& srcView,
141  View<DstMapping, DstBlob>& dstView,
142  bool readOpt,
143  std::size_t threadId = 0,
144  std::size_t threadCount = 1)
145  {
146  // TODO(bgruber): think if we can remove this restriction
147  static_assert(
148  std::is_same_v<typename SrcMapping::RecordDim, typename DstMapping::RecordDim>,
149  "The source and destination record dimensions must be the same");
150  static_assert(
151  std::is_same_v<
152  typename SrcMapping::LinearizeArrayDimsFunctor,
153  typename DstMapping::LinearizeArrayDimsFunctor>,
154  "Source and destination mapping need to use the same array dimensions linearizer");
155  using RecordDim = typename SrcMapping::RecordDim;
156  internal::assertTrivialCopyable<RecordDim>();
157 
158  [[maybe_unused]] static constexpr bool isSrcMB = SrcMapping::blobCount > 1;
159  [[maybe_unused]] static constexpr bool isDstMB = DstMapping::blobCount > 1;
160  static constexpr auto lanesSrc = internal::aosoaLanes<SrcMapping>;
161  static constexpr auto lanesDst = internal::aosoaLanes<DstMapping>;
162 
163  if(srcView.mapping().extents() != dstView.mapping().extents())
164  throw std::runtime_error{"Array dimensions sizes are different"};
165 
166  static constexpr auto srcIsAoSoA = lanesSrc != std::numeric_limits<std::size_t>::max();
167  static constexpr auto dstIsAoSoA = lanesDst != std::numeric_limits<std::size_t>::max();
168 
169  static_assert(srcIsAoSoA || dstIsAoSoA, "At least one of the mappings must be an AoSoA mapping");
170  static_assert(
171  !srcIsAoSoA || std::tuple_size_v<decltype(srcView.storageBlobs)> == 1,
172  "Implementation assumes AoSoA with single blob");
173  static_assert(
174  !dstIsAoSoA || std::tuple_size_v<decltype(dstView.storageBlobs)> == 1,
175  "Implementation assumes AoSoA with single blob");
176 
177  const auto flatSize = product(dstView.mapping().extents());
178 
179  // TODO(bgruber): implement the following by adding additional copy loops for the remaining elements
180  if(!srcIsAoSoA && flatSize % lanesDst != 0)
181  throw std::runtime_error{"Source SoA mapping's total array elements must be evenly divisible by the "
182  "destination AoSoA Lane count."};
183  if(!dstIsAoSoA && flatSize % lanesSrc != 0)
184  throw std::runtime_error{"Destination SoA mapping's total array elements must be evenly divisible by the "
185  "source AoSoA Lane count."};
186 
187  // the same as AoSoA::blobNrAndOffset but takes a flat array index
188  auto mapAoSoA = [](std::size_t flatArrayIndex, auto rc, std::size_t Lanes) LLAMA_LAMBDA_INLINE
189  {
190  const auto blockIndex = flatArrayIndex / Lanes;
191  const auto laneIndex = flatArrayIndex % Lanes;
192  const auto offset = (sizeOf<RecordDim> * Lanes) * blockIndex + offsetOf<RecordDim, decltype(rc)> * Lanes
193  + sizeof(GetType<RecordDim, decltype(rc)>) * laneIndex;
194  return offset;
195  };
196  // the same as SoA::blobNrAndOffset but takes a flat array index
197  auto mapSoA = [&](std::size_t flatArrayIndex, auto rc, bool mb) LLAMA_LAMBDA_INLINE
198  {
199  const auto blob = mb * flatRecordCoord<RecordDim, decltype(rc)>;
200  const auto offset = !mb * offsetOf<RecordDim, decltype(rc)> * flatSize
201  + sizeof(GetType<RecordDim, decltype(rc)>) * flatArrayIndex;
202  return NrAndOffset{blob, offset};
203  };
204 
205  auto mapSrc = [&](std::size_t flatArrayIndex, auto rc) LLAMA_LAMBDA_INLINE
206  {
207  if constexpr(srcIsAoSoA)
208  return &srcView.storageBlobs[0][0] + mapAoSoA(flatArrayIndex, rc, lanesSrc);
209  else
210  {
211  const auto [blob, off] = mapSoA(flatArrayIndex, rc, isSrcMB);
212  return &srcView.storageBlobs[blob][off];
213  }
214  };
215  auto mapDst = [&](std::size_t flatArrayIndex, auto rc) LLAMA_LAMBDA_INLINE
216  {
217  if constexpr(dstIsAoSoA)
218  return &dstView.storageBlobs[0][0] + mapAoSoA(flatArrayIndex, rc, lanesDst);
219  else
220  {
221  const auto [blob, off] = mapSoA(flatArrayIndex, rc, isDstMB);
222  return &dstView.storageBlobs[blob][off];
223  }
224  };
225 
226  static constexpr auto l = []
227  {
228  if constexpr(srcIsAoSoA && dstIsAoSoA)
229  return std::gcd(lanesSrc, lanesDst);
230  return std::min(lanesSrc, lanesDst);
231  }();
232  if(readOpt)
233  {
234  // optimized for linear reading
235  constexpr auto srcL = srcIsAoSoA ? lanesSrc : l;
236  const auto elementsPerThread = flatSize / srcL / threadCount * srcL;
237  {
238  const auto start = threadId * elementsPerThread;
239  const auto stop = threadId == threadCount - 1 ? flatSize : (threadId + 1) * elementsPerThread;
240 
241  auto copyLBlock = [&](const std::byte*& threadSrc, std::size_t dstIndex, auto rc) LLAMA_LAMBDA_INLINE
242  {
243  constexpr auto bytes = l * sizeof(GetType<RecordDim, decltype(rc)>);
244  std::memcpy(mapDst(dstIndex, rc), threadSrc, bytes);
245  threadSrc += bytes;
246  };
247  if constexpr(srcIsAoSoA)
248  {
249  auto* threadSrc = mapSrc(start, RecordCoord<>{});
250  for(std::size_t i = start; i < stop; i += lanesSrc)
251  forEachLeafCoord<RecordDim>(
252  [&](auto rc) LLAMA_LAMBDA_INLINE
253  {
254  for(std::size_t j = 0; j < lanesSrc; j += l)
255  copyLBlock(threadSrc, i + j, rc);
256  });
257  }
258  else
259  {
260  forEachLeafCoord<RecordDim>(
261  [&](auto rc) LLAMA_LAMBDA_INLINE
262  {
263  auto* threadSrc = mapSrc(start, rc);
264  for(std::size_t i = start; i < stop; i += l)
265  copyLBlock(threadSrc, i, rc);
266  });
267  }
268  }
269  }
270  else
271  {
272  // optimized for linear writing
273  constexpr auto dstL = dstIsAoSoA ? lanesDst : l;
274  const auto elementsPerThread = flatSize / dstL / threadCount * dstL;
275  {
276  const auto start = threadId * elementsPerThread;
277  const auto stop = threadId == threadCount - 1 ? flatSize : (threadId + 1) * elementsPerThread;
278 
279  auto copyLBlock = [&](std::byte*& threadDst, std::size_t srcIndex, auto rc) LLAMA_LAMBDA_INLINE
280  {
281  constexpr auto bytes = l * sizeof(GetType<RecordDim, decltype(rc)>);
282  std::memcpy(threadDst, mapSrc(srcIndex, rc), bytes);
283  threadDst += bytes;
284  };
285  if constexpr(dstIsAoSoA)
286  {
287  auto* threadDst = mapDst(start, RecordCoord<>{});
288  for(std::size_t i = start; i < stop; i += lanesDst)
289  forEachLeafCoord<RecordDim>(
290  [&](auto rc) LLAMA_LAMBDA_INLINE
291  {
292  for(std::size_t j = 0; j < lanesDst; j += l)
293  copyLBlock(threadDst, i + j, rc);
294  });
295  }
296  else
297  {
298  forEachLeafCoord<RecordDim>(
299  [&](auto rc) LLAMA_LAMBDA_INLINE
300  {
301  auto* threadDst = mapDst(start, rc);
302  for(std::size_t i = start; i < stop; i += l)
303  copyLBlock(threadDst, i, rc);
304  });
305  }
306  }
