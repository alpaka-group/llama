129  template<typename SrcMapping, typename SrcBlob, typename DstMapping, typename DstBlob>
130  void aosoaCommonBlockCopy(
131  const View<SrcMapping, SrcBlob>& srcView,
132  View<DstMapping, DstBlob>& dstView,
133  bool readOpt,
134  std::size_t threadId = 0,
135  std::size_t threadCount = 1)
136  {
137  // TODO(bgruber): think if we can remove this restriction
138  static_assert(
139  std::is_same_v<typename SrcMapping::RecordDim, typename DstMapping::RecordDim>,
140  "The source and destination record dimensions must be the same");
141  static_assert(
142  std::is_same_v<
143  typename SrcMapping::LinearizeArrayDimsFunctor,
144  typename DstMapping::LinearizeArrayDimsFunctor>,
145  "Source and destination mapping need to use the same array dimensions linearizer");
146  using RecordDim = typename SrcMapping::RecordDim;
147  internal::assertTrivialCopyable<RecordDim>();
148 
149  static constexpr bool MBSrc = SrcMapping::blobCount > 1;
150  static constexpr bool MBDst = DstMapping::blobCount > 1;
151  static constexpr auto LanesSrc = internal::aosoaLanes<SrcMapping>;
152  static constexpr auto LanesDst = internal::aosoaLanes<DstMapping>;
153 
154  if(srcView.mapping().arrayDims() != dstView.mapping().arrayDims())
155  throw std::runtime_error{"Array dimensions sizes are different"};
156 
157  static constexpr auto srcIsAoSoA = LanesSrc != std::numeric_limits<std::size_t>::max();
158  static constexpr auto dstIsAoSoA = LanesDst != std::numeric_limits<std::size_t>::max();
159 
160  static_assert(srcIsAoSoA || dstIsAoSoA, "At least one of the mappings must be an AoSoA mapping");
161  static_assert(
162  !srcIsAoSoA || decltype(srcView.storageBlobs)::rank == 1,
163  "Implementation assumes AoSoA with single blob");
164  static_assert(
165  !dstIsAoSoA || decltype(dstView.storageBlobs)::rank == 1,
166  "Implementation assumes AoSoA with single blob");
167 
168  const auto arrayDims = dstView.mapping().arrayDims();
169  const auto flatSize
170  = std::reduce(std::begin(arrayDims), std::end(arrayDims), std::size_t{1}, std::multiplies<>{});
171 
172  // TODO(bgruber): implement the following by adding additional copy loops for the remaining elements
173  if(!srcIsAoSoA && flatSize % LanesDst != 0)
174  throw std::runtime_error{"Source SoA mapping's total array elements must be evenly divisible by the "
175  "destination AoSoA Lane count."};
176  if(!dstIsAoSoA && flatSize % LanesSrc != 0)
177  throw std::runtime_error{"Destination SoA mapping's total array elements must be evenly divisible by the "
178  "source AoSoA Lane count."};
179 
180  // the same as AoSoA::blobNrAndOffset but takes a flat array index
181  auto mapAoSoA = [](std::size_t flatArrayIndex, auto coord, std::size_t Lanes) LLAMA_LAMBDA_INLINE
182  {
183  const auto blockIndex = flatArrayIndex / Lanes;
184  const auto laneIndex = flatArrayIndex % Lanes;
185  const auto offset = (sizeOf<RecordDim> * Lanes) * blockIndex + offsetOf<RecordDim, decltype(coord)> * Lanes
186  + sizeof(GetType<RecordDim, decltype(coord)>) * laneIndex;
187  return offset;
188  };
189  // the same as SoA::blobNrAndOffset but takes a flat array index
190  auto mapSoA = [&](std::size_t flatArrayIndex, auto coord, bool mb) LLAMA_LAMBDA_INLINE
191  {
192  const auto blob = mb * flatRecordCoord<RecordDim, decltype(coord)>;
193  const auto offset = !mb * offsetOf<RecordDim, decltype(coord)> * flatSize
194  + sizeof(GetType<RecordDim, decltype(coord)>) * flatArrayIndex;
195  return NrAndOffset{blob, offset};
196  };
197 
198  auto mapSrc = [&srcView, &mapAoSoA, &mapSoA](std::size_t flatArrayIndex, auto coord) LLAMA_LAMBDA_INLINE
199  {
200  if constexpr(srcIsAoSoA)
201  return &srcView.storageBlobs[0][0] + mapAoSoA(flatArrayIndex, coord, LanesSrc);
202  else
203  {
204  const auto [blob, off] = mapSoA(flatArrayIndex, coord, MBSrc);
205  return &srcView.storageBlobs[blob][off];
206  }
207  };
208  auto mapDst = [&dstView, &mapAoSoA, &mapSoA](std::size_t flatArrayIndex, auto coord) LLAMA_LAMBDA_INLINE
209  {
210  if constexpr(dstIsAoSoA)
211  return &dstView.storageBlobs[0][0] + mapAoSoA(flatArrayIndex, coord, LanesDst);
212  else
213  {
214  const auto [blob, off] = mapSoA(flatArrayIndex, coord, MBDst);
215  return &dstView.storageBlobs[blob][off];
216  }
217  };

219  static constexpr auto L = []
220  {
221  if constexpr(srcIsAoSoA && dstIsAoSoA)
222  return std::gcd(LanesSrc, LanesDst);
223  return std::min(LanesSrc, LanesDst);
224  }();
225  if(readOpt)
226  {
227  // optimized for linear reading
228  constexpr auto srcL = srcIsAoSoA ? LanesSrc : L;
229  const auto elementsPerThread = flatSize / srcL / threadCount * srcL;
230  {
231  const auto start = threadId * elementsPerThread;
232  const auto stop = threadId == threadCount - 1 ? flatSize : (threadId + 1) * elementsPerThread;
233 
234  auto copyLBlock
235  = [&](const std::byte*& threadSrc, std::size_t dstIndex, auto coord) LLAMA_LAMBDA_INLINE
236  {
237  constexpr auto bytes = L * sizeof(GetType<RecordDim, decltype(coord)>);
238  std::memcpy(mapDst(dstIndex, coord), threadSrc, bytes);
239  threadSrc += bytes;
240  };
241  if constexpr(srcIsAoSoA)
242  {
243  auto* threadSrc = mapSrc(start, RecordCoord<>{});
244  for(std::size_t i = start; i < stop; i += LanesSrc)
245  forEachLeaf<RecordDim>(
246  [&](auto coord) LLAMA_LAMBDA_INLINE
247  {
248  for(std::size_t j = 0; j < LanesSrc; j += L)
249  copyLBlock(threadSrc, i + j, coord);
250  });
251  }
252  else
253  {
254  forEachLeaf<RecordDim>(
255  [&](auto coord) LLAMA_LAMBDA_INLINE
256  {
257  auto* threadSrc = mapSrc(start, coord);
258  for(std::size_t i = start; i < stop; i += L)
259  copyLBlock(threadSrc, i, coord);
260  });
261  }
262  }
263  }
264  else
265  {
266  // optimized for linear writing
267  constexpr auto dstL = dstIsAoSoA ? LanesDst : L;
268  const auto elementsPerThread = flatSize / dstL / threadCount * dstL;
269  {
270  const auto start = threadId * elementsPerThread;
271  const auto stop = threadId == threadCount - 1 ? flatSize : (threadId + 1) * elementsPerThread;
272 
273  auto copyLBlock = [&](std::byte*& threadDst, std::size_t srcIndex, auto coord) LLAMA_LAMBDA_INLINE
274  {
275  constexpr auto bytes = L * sizeof(GetType<RecordDim, decltype(coord)>);
276  std::memcpy(threadDst, mapSrc(srcIndex, coord), bytes);
277  threadDst += bytes;
278  };
279  if constexpr(dstIsAoSoA)
280  {
281  auto* threadDst = mapDst(start, RecordCoord<>{});
282  for(std::size_t i = start; i < stop; i += LanesDst)
283  forEachLeaf<RecordDim>(
284  [&](auto coord) LLAMA_LAMBDA_INLINE
285  {
286  for(std::size_t j = 0; j < LanesDst; j += L)
287  copyLBlock(threadDst, i + j, coord);
288  });
289  }
290  else
291  {
292  forEachLeaf<RecordDim>(
293  [&](auto coord) LLAMA_LAMBDA_INLINE
294  {
295  auto* threadDst = mapDst(start, coord);
296  for(std::size_t i = start; i < stop; i += L)
297  copyLBlock(threadDst, i, coord);
298  });
299  }
300  }
